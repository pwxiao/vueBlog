#include <stdio.h>

const int N = 2022, k = 10;

long long pd[N + 1][k + 1]{1};

int main() {
    for (int i = 1; i <= N; ++i)
        for (int j = 1; j <= k; ++j)
            if (i >= j) pd[i][j] = pd[i - j][j] + pd[i - j][j - 1];
    printf("%lld", pd[N][k]);
}

python
from collections import deque

def solve_maze(maze):
    start = find_start(maze)
    end = find_exit(maze)
    queue = deque([(start, "")])
    visited = set()

    while queue:
        curr, path = queue.popleft()
        row, col = curr

        if curr == end:
            return path

        if curr in visited:
            continue

        visited.add(curr)

        # 尝试向上移动
        if row > 0 and maze[row-1][col] == 0:
            queue.append(((row-1, col), path + "w"))

        # 尝试向下移动
        if row < len(maze) - 1 and maze[row+1][col] == 0:
            queue.append(((row+1, col), path + "s"))

        # 尝试向左移动
        if col > 0 and maze[row][col-1] == 0:
            queue.append(((row, col-1), path + "a"))

        # 尝试向右移动
        if col < len(maze[0]) - 1 and maze[row][col+1] == 0:
            queue.append(((row, col+1), path + "d"))

    return None

def find_start(maze):
    for i in range(len(maze)):
        for j in range(len(maze[0])):
            if maze[i][j] == 0:
                return (i, j)

def find_exit(maze):
    for i in range(len(maze)):
        if maze[i][-1] == 0:
            return (i, len(maze[0])-1)
